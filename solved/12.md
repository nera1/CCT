import java.util.*;

class Solution {
    public int solution(int n, int k, int[] enemy) {
        int answer = 0;
        PriorityQueue<Integer> MaxPQ = new PriorityQueue<>((a,b)->b-a);
        PriorityQueue<Integer> MinPQ = new PriorityQueue<>((a,b)->a-b);
        int total = 0;
        int bigTotal = 0;
        for(int i = 0; i < enemy.length; i++) {
            int val = enemy[i];
            MaxPQ.add(val);
            total += val;
            if(MinPQ.size() < k) {
                MinPQ.add(val);
                bigTotal += val;
            } else if(MinPQ.size() == k) {
                int min = MinPQ.peek();
                if(min < val) {
                    bigTotal -= MinPQ.poll();
                    bigTotal += val;
                    MinPQ.add(val);
                }
            }
            if(total - bigTotal > n) {
                return i;
            }
        }
        return enemy.length;
    }
}
