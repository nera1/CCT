```java
import java.io.*;
import java.util.*;

public class Main {
    public static void printer(int[][] map) {
        for (int i = 0; i < map.length; i++) {
            for (int j = 0; j < map[i].length; j++) {
                System.out.print(map[i][j]);
            }
            System.out.println("");
        }
        System.out.println("---------------------------");
    }

    public static int melt(int[][] map) {
        int count = 0;
        int[][] check = new int[map.length][map[0].length];
        List<int[]> melted = new ArrayList<>();
        Queue<int[]> queue = new LinkedList<>();
        queue.add(new int[] { 0, 0 });
        int[][] move = new int[][] { { 0, 1 }, { 1, 0 }, { -1, 0 }, { 0, -1 } };
        check[0][0] = 1;
        while (queue.size() > 0) {
            int[] pos = queue.poll();
            int r = pos[0];
            int c = pos[1];
            if (map[r][c] == 1) {
                melted.add(new int[] { r, c });
                continue;
            }
            for (int[] m : move) {
                int nextR = r + m[0];
                int nextC = c + m[1];
                if (nextR < 0 || nextC < 0 || nextR >= map.length || nextC >= map[0].length) {
                    continue;
                }
                if (check[nextR][nextC] == 1) {
                    continue;
                }
                check[nextR][nextC] = 1;
                queue.add(new int[] { nextR, nextC });
            }
        }

        for (int[] pos : melted) {
            map[pos[0]][pos[1]] = 0;
            count++;
        }

        return count;
    }

    public static void solution() throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        String[] line1 = br.readLine().split(" ");
        int row = Integer.parseInt(line1[0]);
        int col = Integer.parseInt(line1[1]);
        int[][] map = new int[row][col];
        int total = 0;

        for (int i = 0; i < row; i++) {
            String line = br.readLine();
            String[] nums = line.split(" ");
            for (int j = 0; j < col; j++) {
                map[i][j] = Integer.parseInt(nums[j]);
                if (map[i][j] == 1) {
                    total++;
                }
            }
        }

        int sum = 0;
        int count = 0;
        int last = 0;

        while (sum < total) {
            last = total - sum;
            int result = melt(map);
            sum += result;
            count++;
        }

        System.out.println(count);
        System.out.println(last);

        br.close();
    }

    public static void main(String[] args) throws IOException {
        solution();
    }
}
```
