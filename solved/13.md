```JAVA
import java.util.*;
import java.lang.Math;
class Solution {
    private long min = (long)Integer.MAX_VALUE;
    public int solution(int[] queue1, int[] queue2) {
        Queue<Long> q1 = new LinkedList<>();
        long q1Sum = 0;
        Queue<Long> q2 = new LinkedList<>();
        long q2Sum = 0;
        for(int i = 0; i < queue1.length; i++) {
            long q1i = (long)queue1[i];
            long q2i = (long)queue2[i];
            q1Sum += q1i;
            q2Sum += q2i;
            q1.add(q1i);
            q2.add(q2i);
        }
        long max = 2*(q1.size() + q2.size());
        long sum = q2Sum + q1Sum;
        long goal = (sum)/2;
        if(sum%2 == 1) {
            return -1;
        }
        int count = 0;
        int i = 0;
        for(;count <= max;) {
            if(q1Sum == goal) {
                break;
            }
            if(q1.size() > 0 && q1Sum > goal) {
                long q1top = q1.poll();
                q1Sum -= q1top;
                q2.add(q1top);
                q2Sum += q1top;
                count++;
            } else if(q2.size() > 0 && q2Sum > goal) {
                long q2top = q2.poll();
                q2Sum -= q2top;
                q1.add(q2top);
                q1Sum += q2top;
                count++;
            } else {
                break;
            }
        }
        return count > max ? -1 : count;
    }
}
```
