import java.io.*;
import java.util.*;

public class Main {
    public static int pour(int[][] map, int[] pos) {
        int count = 0;
        int row = pos[0];
        int col = pos[1];
        map[row][col] = 2;
        count++;
        int l = col - 1;
        while(l >= 0 && map[row][l] == 0) {
            map[row][l--] = 2;
            count++;
        }
        int r = col + 1;
        while(r < map[0].length && map[row][r] == 0) {
            map[row][r++] = 2;
            count++;
        }
        return count;
    }
    public static boolean findSide(int[][] map, int[] pos) {
        int col = pos[1];
        int row = pos[0];
        boolean left = false;
        //left
        int l = col;
        while(l >= 0) {
            if(map[row][l] == 1) {
                left = true;
                break;
            } else {
                l--;
            }
        }
        boolean right = false;
        int r = col;
        while(r < map[0].length) {
            if(map[row][r] == 1) {
                right = true;
                break;
            } else {
                r++;
            }
        }
        //right
        return left & right;
    }
    public static void solution() throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        String str = br.readLine();
        String[] wh = str.split(" ");
        int answer = 0;

        int row = Integer.parseInt(wh[0]);
        int col = Integer.parseInt(wh[1]);

        int[][] map = new int[row][col];

        String line = br.readLine();
        String[] nums = line.split(" ");

        for(int i = 0; i < nums.length; i++) {
            int count = Integer.parseInt(nums[i]);
            for(int j = row - 1; j >= row - count; j--) {
                  map[j][i] = 1;
            }
        }

        for(int i = row - 1; i >= 0; i--) {
            for(int j = 0; j < col; j++) {
               if(map[i][j] >= 1) {
                   continue;
               }
               int[] pair = new int[]{i, j};
               if(findSide(map, pair)) {
                   answer += pour(map, pair);
               }
            }
        }

        System.out.println(answer);

        br.close();
    }

    public static void main(String[] args) throws IOException {
        solution();
    }
}
