import java.util.*;

class Solution {
    public int strTolong(String date) {
        int value = 0;
        String[] strVal = date.split("\\.");
        String y = strVal[0];
        String m = strVal[1];
        String d = strVal[2];
        value += (new Integer(y)) * 12 * 28;
        value += (new Integer(m)) * 28;
        value += (new Integer(d));
        return value;
    }
    public int[] solution(String today, String[] terms, String[] privacies) {
        int tVal = strTolong(today);
        Map<String, Integer> map = new HashMap<>();
        for(String t : terms) {
            String[] tv = t.split(" ");
            map.put(tv[0], Integer.parseInt(tv[1]));
        }
        int count = 1;
        List<Integer> list = new ArrayList<>();
        for(String p : privacies) {
            String[] pv = p.split(" ");
            int v = strTolong(pv[0]);
            v += map.get(pv[1]) * 28;
            if(v <= tVal) {
                list.add(count);
            }
            count++;
        }
        return list.stream().mapToInt(Integer::intValue).toArray();
    }
}

public class SplitExample {
    public static void main(String[] args) {
        String text = "Hello.World*Java+Regex?Test^Begin$End|Choice[Array]{Curly}(Group)";

        // 역슬래시 (\) 기준으로 분할
        String[] splitByBackslash = text.split("\\\\");

        // 점 (.) 기준으로 분할
        String[] splitByDot = text.split("\\.");

        // 애스터리스크 (*) 기준으로 분할
        String[] splitByAsterisk = text.split("\\*");

        // 플러스 (+) 기준으로 분할
        String[] splitByPlus = text.split("\\+");

        // 물음표 (?) 기준으로 분할
        String[] splitByQuestionMark = text.split("\\?");

        // 캐럿 (^) 기준으로 분할
        String[] splitByCaret = text.split("\\^");

        // 달러 기호 ($) 기준으로 분할
        String[] splitByDollar = text.split("\\$");

        // 파이프 (|) 기준으로 분할
        String[] splitByPipe = text.split("\\|");

        // 대괄호 ([]) 기준으로 분할 (열고 닫는 대괄호 모두 포함)
        String[] splitBySquareBrackets = text.split("\\[|\\]");

        // 중괄호 ({}) 기준으로 분할 (열고 닫는 중괄호 모두 포함)
        String[] splitByCurlyBraces = text.split("\\{|\\}");

        // 소괄호 (()) 기준으로 분할 (열고 닫는 소괄호 모두 포함)
        String[] splitByParentheses = text.split("\\(|\\)");

        // 결과 출력
        printArray("Split by Backslash:", splitByBackslash);
        printArray("Split by Dot:", splitByDot);
        printArray("Split by Asterisk:", splitByAsterisk);
        printArray("Split by Plus:", splitByPlus);
        printArray("Split by Question Mark:", splitByQuestionMark);
        printArray("Split by Caret:", splitByCaret);
        printArray("Split by Dollar:", splitByDollar);
        printArray("Split by Pipe:", splitByPipe);
        printArray("Split by Square Brackets:", splitBySquareBrackets);
        printArray("Split by Curly Braces:", splitByCurlyBraces);
        printArray("Split by Parentheses:", splitByParentheses);
    }

    private static void printArray(String message, String[] array) {
        System.out.println(message);
        for (String part : array) {
            System.out.println(part);
        }
        System.out.println();
    }
}
